Custom pipeline for NER

Lets create a custom component for NER to detect movies in a text.

# custom pipeline for NER
from spacy.matcher import PhraseMatcher
from spacy.tokenizer import Tokenizer

nlp = spacy.load("en_core_web_sm")

bond_movies = ["Casino Royale", "Quantum of Solace", "Skyfall", "Spectre", "No Time To Die"]
movie_matcher = PhraseMatcher(nlp.vocab)

movie_patterns = [nlp(movie) for movie in bond_movies]
# using a pipe is faster
movie_patterns_2 = nlp.tokenizer.pipe(bond_movies)

movie_matcher.add("Bond", movie_patterns)

# process a text
text = "I wached No Time To Die last night in India. Great movie!"
doc = nlp(text)

# perform ner on the text. Also print the type of entity
for match_id, start, end in movie_matcher(doc):
    span = doc[start:end]
    print(span.text)
